** start of script.js **

function convertMarkdown() {
    const input = document.getElementById("markdown-input").value;
    const htmlOutput = document.getElementById("html-output");
    const preview = document.getElementById("preview");

    htmlOutput.textContent = ''; // Clear previous HTML
    preview.innerHTML = '';      // Clear previous preview

    const lines = input.split('\n');
    let html = '';

    lines.forEach(line => {
        let convertedLine = line;

        // Headings
        if (/^### (.+)/.test(convertedLine)) {
            convertedLine = convertedLine.replace(/^### (.+)/, '<h3>$1</h3>');
        } else if (/^## (.+)/.test(convertedLine)) {
            convertedLine = convertedLine.replace(/^## (.+)/, '<h2>$1</h2>');
        } else if (/^# (.+)/.test(convertedLine)) {
            convertedLine = convertedLine.replace(/^# (.+)/, '<h1>$1</h1>');
        }
        // Blockquotes
        else if (/^>\s+(.+)/.test(convertedLine)) {
            convertedLine = convertedLine.replace(/^>\s+(.+)/, '<blockquote>$1</blockquote>');
        }

        // Bold (**text** or __text__)
        convertedLine = convertedLine.replace(/(\*\*|__)(.*?)\1/g, '<strong>$2</strong>');

        // Italic (*text* or _text_)
        convertedLine = convertedLine.replace(/(\*|_)(.*?)\1/g, '<em>$2</em>');

        // Images ![alt](src)
        convertedLine = convertedLine.replace(/!\[(.*?)\]\((.*?)\)/g, '<img alt="$1" src="$2">');

        // Links [text](url)
        convertedLine = convertedLine.replace(/\[(.*?)\]\((.*?)\)/g, '<a href="$2">$1</a>');

        // Append to raw HTML string
        html += convertedLine;

        // Append each converted line as DOM element
        const tempDiv = document.createElement('div');
        tempDiv.innerHTML = convertedLine;
        while (tempDiv.firstChild) {
            preview.appendChild(tempDiv.firstChild);
        }
    });

    htmlOutput.textContent = html; // Show raw HTML
    return html;
}

// Input event listener
document.getElementById("markdown-input").addEventListener("input", convertMarkdown);


** end of script.js **
